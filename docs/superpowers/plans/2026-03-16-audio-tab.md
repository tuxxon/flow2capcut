### Task 2: useAudioImport — review 데이터 로드/저장

**Files:**

- [ ] **Step 1: review state 및 load/save 함수 추가**

`useAudioImport` 훅에 추가:

```javascript
const [audioReviews, setAudioReviews] = useState({})

// .audio_review.json 경로 계산
const getReviewPath = useCallback((folderPath) => {
  if (!folderPath) return null
  return `${folderPath}/.audio_review.json`
}, [])

// review 데이터 로드
const loadReviews = useCallback(async (folderPath) => {
  const reviewPath = getReviewPath(folderPath)
  if (!reviewPath) return {}
  try {
    const result = await window.electronAPI?.readFileAbsolute({ filePath: reviewPath })
    if (result?.success && result.data) {
      // readFileAbsolute returns data URL, extract JSON
      const base64 = result.data.split(',')[1]
      const json = atob(base64)
      const reviews = JSON.parse(json)
      setAudioReviews(reviews)
      return reviews
    }
  } catch (e) { /* no review file yet */ }
  return {}
}, [getReviewPath])

// review 데이터 저장
const saveReview = useCallback(async (folderPath, relativePath, review) => {
  const reviewPath = getReviewPath(folderPath)
  if (!reviewPath) return
  const updated = {
    ...audioReviews,
    [relativePath]: review
      ? { status: 'flagged', reason: review.reason, flaggedAt: new Date().toISOString() }
      : undefined  // null이면 마크 해제
  }
  // undefined 키 제거
  Object.keys(updated).forEach(k => { if (!updated[k]) delete updated[k] })
  setAudioReviews(updated)
  await window.electronAPI?.writeFileAbsolute({
    filePath: reviewPath,
    content: JSON.stringify(updated, null, 2)
  })
}, [audioReviews, getReviewPath])
```

```jsx
const timelineItems = useMemo(() => {
  const items = []
  // Voice files
  if (audioPackage.voices) {
    for (const v of audioPackage.voices) {

        items.push({
          type: 'voice',
          timecodeMs: f.timecodeMs,

        })

    }
  }
  // SFX files (with timecodes from sfxTimecodes)
  if (audioPackage.sfx) {
    for (const cat of audioPackage.sfx) {
      for (const f of cat.files) {
        items.push({
          type: 'sfx',
          timecodeMs: f.timecodeMs || null,
          character: cat.category,
          filename: f.filename,
          path: f.path,
          relativePath: f.path.replace(audioPackage.folderPath + '/', '')
        })
      }
    }
  }
  // Sort by timecode (nulls at end)
  items.sort((a, b) => (a.timecodeMs ?? Infinity) - (b.timecodeMs ?? Infinity))
  return items
}, [audioPackage])

// SRT 매칭
const findMatchingSrt = useCallback((timecodeMs) => {
  if (!srtEntries?.length || timecodeMs == null) return null
  return srtEntries.find(e => timecodeMs >= e.startMs && timecodeMs <= e.endMs)
    || srtEntries.reduce((closest, e) => {
        if (e.endMs <= timecodeMs) {
          if (!closest || e.endMs > closest.endMs) return e
        }
        return closest
      }, null)
}, [srtEntries])
```

```jsx
import { useState, useRef, useEffect } from 'react'

export default function AudioFlagPopover({ target, existingReview, onSave, onRemove, onClose }) {
  const [reason, setReason] = useState(existingReview?.reason || '')
  const inputRef = useRef(null)

  useEffect(() => {
    inputRef.current?.focus()
  }, [])

  return (
    <div className="flag-popover-backdrop" onClick={onClose}>
      <div className="flag-popover" onClick={e => e.stopPropagation()}
        style={{ top: target.y, left: target.x }}>
        <div className="flag-popover-title">
          ⚠️ {target.filename}
        </div>
        <textarea
          ref={inputRef}
          className="flag-reason-input"
          placeholder="부적합 사유 (예: 초인종 소리, 조선시대에 안 맞음)"
          value={reason}
          onChange={e => setReason(e.target.value)}
          rows={3}
        />
        <div className="flag-popover-actions">
          {existingReview && (
            <button className="btn btn-danger-outline" onClick={onRemove}>
              마크 해제
            </button>
          )}
          <button className="btn btn-primary" onClick={() => onSave(reason)}
            disabled={!reason.trim()}>
            저장
          </button>
        </div>
      </div>
    </div>
  )
}
```
